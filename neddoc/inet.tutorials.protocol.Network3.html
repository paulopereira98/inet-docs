Source code

network Network3
{
    submodules:
        client: ClientHost3 {
            @display("p=100,100");
        }
        server: ServerHost3 {
            @display("p=200,100");
        }
    connections allowunconnected:
        client.out --> {  delay = 1us; } --> server.in;
}
