network NetworkPathSimpleShowcase
{
    @display("bgb=600,250");
    submodules:
        pathVisualizer: NetworkRouteVisualizer {
            parameters:
                @display("p=100,50");
        }
        configurator: Ipv4NetworkConfigurator {
            parameters:
                @display("p=100,150");
        }
        source: StandardHost {
            @display("p=200,125");
        }
        destination: StandardHost {
            @display("p=500,125");
        }

    connections:
        source.ethg++ <--> Eth100M <--> destination.ethg++;
}
